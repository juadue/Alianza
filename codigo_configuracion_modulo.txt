SoftwareSerial Bluetooth(10, 11); // RX, TX


void setup() {
  // put your setup code here, to run once:
Serial.begin(38400);  
Serial.println("listo");
Bluetooth.begin(38400);
}

void loop() {

 }

   }


  
}
